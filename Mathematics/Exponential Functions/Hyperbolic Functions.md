$$
\begin{align}
\text{cosh}(x) &= {e^x + e^{-x} \over 2} \\
\text{cosh}(-x)&= {e^{-x} + e^{x}\over 2} \\
\text{cosh}(-x)&=\text{cosh}(x)
\end{align}
$$
$$
\begin{align}
\text{cosh}(0)&=1 \\
\lim_\limits{x \to \infty}({e^x+e^{-x} \over 2})&={e^x \over 2} \\
\lim_\limits{x \to - \infty}({e^x+e^{-x} \over 2})&={e^{-x} \over 2}
\end{align}
$$

$$
\begin{align}
\text{sinh}(x) &= {e^x - e^{-x} \over 2} \\
\text{sinh}(-x)&= {e^{-x} - e^{x}\over 2} = - {e^x - e^{-x} \over 2}\\
\text{sinh}(-x)&=-\text{sinh}(x)
\end{align}
$$

$$
\begin{align}
\text{sinh}(0)&=0 \\
\lim_\limits{x \to \infty}({e^x-e^{-x} \over 2})&={e^x \over 2} \\
\lim_\limits{x \to - \infty}({e^x-e^{-x} \over 2})&=-{e^{-x} \over 2}
\end{align}
$$
